\begin{frame}[fragile]{Ufuncs}
В Python:
\begin{verbatim}
>>> lst = [[1, 2, 3], [4, 5, 6]]
>>>   for row_idx, row_val in enumerate(lst):

>>> lst = [[1, 2, 3], [4, 5, 6]]
